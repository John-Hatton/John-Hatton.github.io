174std::vector<bool> TwoToFourLineDecoder::currentState() {
175
176
177 selectInputNotX->setInput(inputX);
178 selectInputNotX->answer();
179 selectInputNotY->setInput(inputY);
180 selectInputNotY->answer();
181
182 selectInputIBX->setInput(selectInputNotX->answer());
183 selectInputIBY->setInput(selectInputNotY->answer());
184
185
186
187 // Quad Nand Population
188
189 triNandData0->setInputX(selectInputNotX->answer());
190 triNandData0->setInputY(selectInputNotY->answer());
191 triNandData0->setInputZ(writeEnable);
192 lineOut0 = triNandData0->answer();
193
194 triNandData1->setInputX(selectInputIBX->answer());
195 triNandData1->setInputY(selectInputNotY->answer());
196 triNandData1->setInputZ(writeEnable);
197 lineOut1 = triNandData1->answer();
198
199 triNandData2->setInputX(selectInputNotX->answer());
200 triNandData2->setInputY(selectInputIBY->answer());
201 triNandData2->setInputZ(writeEnable);
202 lineOut2 = triNandData2->answer();
203
204 triNandData3->setInputX(selectInputIBX->answer());
205 triNandData3->setInputY(selectInputIBY->answer());
206 triNandData3->setInputZ(writeEnable);
207 lineOut3 = triNandData3->answer();
208
209
210 // Also remember, according to the datasheet, if G1 is EVER 0, we should always get all 1's (or 0's)
211
212 if (!writeEnable)
213 {
214 return {false,false,false,false};
215 }
216
217 // Remember, we return inverted outputs, because the original datasheet has
218 // reverse polarity. We want to maintain normal polarity, and by doing so, we won't need
219 // a lot of reverse logic with the other circuits. Perhaps this isn't wisdom, since most
220 // of these drawings are designed to go together, but I have other drawings, not from TI. Lol
221 return {!lineOut0,!lineOut1,!lineOut2,!lineOut3};
222}
