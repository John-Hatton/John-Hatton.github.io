293std::vector<bool> EightBitAdder::answer() {
294 bitOneFullAdder->setCarryIn(carryIn); // Remember we have a Carry-In!
295 bitOneFullAdder->setInputX(inputX1);
296 bitOneFullAdder->setInputY(inputY1);
297 bitOneFullAdder->answer(); // Must call answer(), to populate sum variable
298 sumOneOut = bitOneFullAdder->getSum();
299
300 bitTwoFullAdder->setCarryIn(bitOneFullAdder->getCarryOut());
301 bitTwoFullAdder->setInputX(inputX2);
302 bitTwoFullAdder->setInputY(inputY2);
303 bitTwoFullAdder->answer();
304 sumTwoOut = bitTwoFullAdder->getSum();
305
306 bitThreeFullAdder->setCarryIn(bitTwoFullAdder->getCarryOut());
307 bitThreeFullAdder->setInputX(inputX3);
308 bitThreeFullAdder->setInputY(inputY3);
309 bitThreeFullAdder->answer();
310 sumThreeOut = bitThreeFullAdder->getSum();
311
312 bitFourFullAdder->setCarryIn(bitThreeFullAdder->getCarryOut());
313 bitFourFullAdder->setInputX(inputX4);

316 sumFourOut = bitFourFullAdder->getSum();

320 bitFiveFullAdder->setCarryIn(bitFourFullAdder->getCarryOut());

323 bitFiveFullAdder->answer();
324 sumFiveOut = bitFiveFullAdder->getSum();
325
326 bitSixFullAdder->setCarryIn(bitFiveFullAdder->getCarryOut());
327 bitSixFullAdder->setInputX(inputX6);
328 bitSixFullAdder->setInputY(inputY6);
329 bitSixFullAdder->answer();
330 sumSixOut = bitSixFullAdder->getSum();
331
332 bitSevenFullAdder->setCarryIn(bitSixFullAdder->getCarryOut());
333 bitSevenFullAdder->setInputX(inputX7);

336 sumSevenOut = bitSevenFullAdder->getSum();

338 bitEightFullAdder->setCarryIn(bitSevenFullAdder->getCarryOut());
