149std::vector<bool> FourBitAdder::answer() {
150
151 bitOneFullAdder->setInputX(inputX1);
152 bitOneFullAdder->setInputY(inputY1);
153 bitOneFullAdder->answer(); // Must call answer(), to populate sum variable
154 sumOneOut = bitOneFullAdder->getSum();
155 bitTwoFullAdder->setCarryIn(bitOneFullAdder->getCarryOut());
156 bitTwoFullAdder->setInputX(inputX2);
157 bitTwoFullAdder->setInputY(inputY2);
158 bitTwoFullAdder->answer();
159 sumTwoOut = bitTwoFullAdder->getSum();
160 bitThreeFullAdder->setCarryIn(bitTwoFullAdder->getCarryOut());
161 bitThreeFullAdder->setInputX(inputX3);
162 bitThreeFullAdder->setInputY(inputY3);
163 bitThreeFullAdder->answer();
164 sumThreeOut = bitThreeFullAdder->getSum();
165 bitFourFullAdder->setCarryIn(bitTwoFullAdder->getCarryOut());
166 bitFourFullAdder->setInputX(inputX4);
167 bitFourFullAdder->setInputY(inputY4);
168 bitFourFullAdder->answer();
169 sumFourOut = bitFourFullAdder->getSum();
170 carryOut = bitFourFullAdder->getCarryOut();
171
172 return {carryOut,sumFourOut,sumThreeOut,sumTwoOut,sumOneOut};
173}
